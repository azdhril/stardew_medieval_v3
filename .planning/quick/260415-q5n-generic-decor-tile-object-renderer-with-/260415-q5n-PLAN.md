Fields (all private readonly, set in ctor):

- `Rectangle _sourceRect` (tile rect in the tileset atlas — computed by TileMap from local id + columns)
- `Rectangle _destRect` (world-space destination — computed by TileMap from obj.x, obj.y - tileHeight, tileWidth, tileHeight; Tiled tile-object anchor is bottom-left)
- `int _occlusionY` (clamped to [0, sourceRect.Height - 1] in ctor)
- `SpriteEffects _flipEffects` (from GID flip flags — strip with `& 0x1FFFFFFF` after reading)
- `Vector2 _worldAnchor` = new(destRect.Center.X, destRect.Bottom) — bottom-center of the dest rect, used for feet-behind test

Public methods (mirror ResourceNode EXACTLY — same 3-condition test, same split logic, same 0.5f fade):
```csharp
public void DrawBeforePlayer(SpriteBatch sb, PlayerEntity? player)
{
    if (player == null || !ShouldUseFrontOccluder(player))
    {
        sb.Draw(_texture, _destRect, _sourceRect, Color.White, 0f, Vector2.Zero, _flipEffects, 0f);
        return;
    }
    if (_occlusionY > 0)
    {
        var backSrc  = new Rectangle(_sourceRect.X, _sourceRect.Y, _sourceRect.Width, _occlusionY);
        var backDest = new Rectangle(_destRect.X,   _destRect.Y,   _destRect.Width,   _occlusionY);
        sb.Draw(_texture, backDest, backSrc, Color.White, 0f, Vector2.Zero, _flipEffects, 0f);
    }
}

public void DrawAfterPlayer(SpriteBatch sb, PlayerEntity player)
{
    if (!ShouldUseFrontOccluder(player)) return;
    var frontSrc  = new Rectangle(_sourceRect.X, _sourceRect.Y + _occlusionY, _sourceRect.Width, _sourceRect.Height - _occlusionY);
    var frontDest = new Rectangle(_destRect.X,   _destRect.Y   + _occlusionY, _destRect.Width,   _destRect.Height   - _occlusionY);
    sb.Draw(_texture, frontDest, frontSrc, Color.White * 0.5f, 0f, Vector2.Zero, _flipEffects, 0f);
}

private bool ShouldUseFrontOccluder(PlayerEntity player)
{
    var pb = player.CollisionBox;
    bool overlapsH = pb.Right  > _destRect.Left + 8 && pb.Left < _destRect.Right - 8;
    bool overlapsV = pb.Bottom > _destRect.Top + _occlusionY && pb.Top < _destRect.Bottom - 6;
    bool feetBehind = player.GetFootPosition().Y < _worldAnchor.Y - 1f;
    return overlapsH && overlapsV && feetBehind;
}
```

4. **Add `LoadDecorObjects` method** (called from `Load` right after `LoadTriggerObjects()`):
   ```csharp
   private void LoadDecorObjects()
   {
       _decor.Clear();
       var objs = GetObjectGroup("Decor");
       foreach (var obj in objs)
       {
           if (obj.Gid == 0) continue; // skip non-tile objects (rects/points/polys)

           int rawGid = obj.Gid;
           int gid = rawGid & 0x1FFFFFFF;

           // Resolve tileset (same logic as DrawTileByGid)
           int firstGid = 0; TiledTileset? ts = null; Texture2D? tex = null;
           foreach (var kvp in _tilesets)
               if (gid >= kvp.Key && kvp.Key > firstGid)
               { firstGid = kvp.Key; ts = kvp.Value; tex = _tilesetTextures[kvp.Key]; }
           if (ts == null || tex == null) continue;

           int localId = gid - firstGid;
           int tileW = ts.TileWidth, tileH = ts.TileHeight, cols = ts.Columns;
           if (cols <= 0) continue;

           var src = new Rectangle((localId % cols) * tileW, (localId / cols) * tileH, tileW, tileH);

           // Tiled tile-object anchor: bottom-left at (obj.x, obj.y)
           var dest = new Rectangle((int)obj.Bounds.X, (int)obj.Bounds.Y - tileH, tileW, tileH);

           // occlusion_y lookup: per-tile override > fallback tileH/2
           int occY = tileH / 2;
           if (_occlusionYByTileset.TryGetValue(firstGid, out var lookup) && lookup.TryGetValue(localId, out var v))
               occY = v;

           var effects = SpriteEffects.None;
           if ((rawGid & unchecked((int)0x80000000)) != 0) effects |= SpriteEffects.FlipHorizontally;
           if ((rawGid & 0x40000000) != 0) effects |= SpriteEffects.FlipVertically;

           _decor.Add(new DecorOccluder(tex, src, dest, occY, effects));
       }
       if (_decor.Count > 0)
           Console.WriteLine($"[TileMap] Loaded {_decor.Count} decor occluders");
   }
   ```
   Call `LoadDecorObjects()` from `Load()` after `LoadTriggerObjects()`. Update the summary log line at end of `Load` to include decor count.
