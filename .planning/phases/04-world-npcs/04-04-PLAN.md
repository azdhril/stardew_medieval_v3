       private SpriteAtlas _atlas = null!;

           _toast = new Toast();
       }

       public override void LoadContent()
       {
           _pixel = /* obtain 1x1 white texture via Services.Content or shared static */;
           _font = Services.Content.Load<SpriteFont>("DefaultFont");
           // atlas already resolved in ctor
           Console.WriteLine("[ShopOverlayScene] Opened");
       }

       public override void Update(float dt)
       {
           if (_panel.Update(dt, Services.Input, out var toastReq)) {
               Console.WriteLine("[ShopOverlayScene] Closed");
               Services.SceneManager.PopImmediate();
