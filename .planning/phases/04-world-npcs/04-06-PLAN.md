       for (int i = 0; i < VisibleRows; i++)

3. Modify `Update(float dt, InputManager input, out ToastRequest? toast)`:

   Inject mouse handling BEFORE the existing keyboard block. The final `Update` shape:

   ```csharp
   public bool Update(float dt, InputManager input, out ToastRequest? toast)
   {
       toast = null;

       // --- Compute current layout (shared with Draw) ---
       int rows = GetRowCount();
       int visible = Math.Min(rows, VisibleRows);
       int scroll = 0;
       if (_selectedIndex >= VisibleRows) scroll = _selectedIndex - VisibleRows + 1;
       RecomputeLayout(visible, scroll, rows);

       // --- Mouse hit-test ---
       var mp = input.MousePosition;
       _hoveredRow = -1;
       for (int i = 0; i < visible; i++)
       {
           if (_rowRects[i].Contains(mp)) { _hoveredRow = i; break; }
       }

       if (input.IsLeftClickPressed)
       {
           // 1. Close X
           if (_closeRect.Contains(mp)) return true;

           // 2. Click outside panel → close
           if (!_panelRect.Contains(mp)) return true;

           // 3. Tabs
           if (_buyTabRect.Contains(mp))  { if (_tab != Tab.Buy)  { _tab = Tab.Buy;  _selectedIndex = 0; } return false; }
           if (_sellTabRect.Contains(mp)) { if (_tab != Tab.Sell) { _tab = Tab.Sell; _selectedIndex = 0; } return false; }

           // 4. Action button — only on the selected row and only if enabled
           if (_hoveredRow >= 0)
           {
               int absIndex = _hoveredRow + scroll;
               // Row-body click selects the row
               _selectedIndex = absIndex;

               // If the click landed inside the row's action button AND row is enabled, fire action
               if (_actionBtnRects[_hoveredRow].Contains(mp) && IsActionEnabled(absIndex))
               {
                   if (_tab == Tab.Buy) TryBuy(out toast);
                   else TrySell(out toast);
               }
               return false;
           }
       }

       // --- Existing keyboard block (unchanged) ---
       if (input.IsKeyPressed(Keys.Escape)) return true;
       if (input.IsKeyPressed(Keys.Tab))
       {
           _tab = _tab == Tab.Buy ? Tab.Sell : Tab.Buy;
           _selectedIndex = 0;
           return false;
       }

       if (rows > 0)
       {
           if (input.IsKeyPressed(Keys.Down)) _selectedIndex = (_selectedIndex + 1) % rows;
           if (input.IsKeyPressed(Keys.Up))   _selectedIndex = (_selectedIndex - 1 + rows) % rows;
       }
       else
       {
           _selectedIndex = 0;
       }

       if (input.IsKeyPressed(Keys.Enter))
       {
           if (_tab == Tab.Buy) TryBuy(out toast);
           else TrySell(out toast);
       }

       return false;
   }
   ```

   Notes on correctness:
   - `IsActionEnabled(absIndex)` is re-used from the existing private helper (line 365).
   - Row-body click ALWAYS updates `_selectedIndex`; if the click ALSO landed inside the action button AND the row is enabled, the transaction fires on the same click. This matches typical shop UX (click the Buy button = buy). Clicking anywhere else on the row just selects.
   - Note that `_hoveredRow` being the VISIBLE-row index (0..visible-1), while `_selectedIndex` is the ABSOLUTE index (0..rows-1). Conversion is `absIndex = _hoveredRow + scroll`.
