         int hotbarSlotSize = 50;

         int hotbarWidth = 8 * hotbarSlotSize + 7 * 4; // 8 slots with 4px padding

         ```csharp
         int xpBarH = 10;
         int xpBarY = hotbarTopY - xpBarH - 6;
         int xpBarX = hotbarStartX - 8; // align with panel padding
         int xpBarW = hotbarWidth + 16;
         
         // Draw XP bar background
         if (_theme?.XPBarBg != null)
         {
             float xpScale = (float)xpBarH / _theme.XPBarBg.Height;
             int bgW = (int)(_theme.XPBarBg.Width * xpScale);
             // Scale to fit hotbar width instead:
             spriteBatch.Draw(_theme.XPBarBg, new Rectangle(xpBarX, xpBarY, xpBarW, xpBarH), Color.White);
         }
         
         // Draw XP bar fill
         float xpFill = _progression != null && _progression.XPToNext > 0
             ? (float)_progression.XP / _progression.XPToNext : 0f;
         xpFill = MathHelper.Clamp(xpFill, 0f, 1f);
         if (_theme?.XPBarFill != null && xpFill > 0)
         {
             int fillW = (int)(xpBarW * xpFill);
             int srcW = (int)(_theme.XPBarFill.Width * xpFill);
             spriteBatch.Draw(_theme.XPBarFill, new Rectangle(xpBarX, xpBarY, fillW, xpBarH),

         int level = _progression?.Level ?? 1;

         ```
